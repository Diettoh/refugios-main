- Export a `mockQuery` function that replaces the real `query()` from `src/db/client.js`
- Accept a `responses` map (SQL text fragment -> result object) to control what each query returns
- Export a `resetMock()` function to clear recorded calls between tests
- Export a `getCalls()` function to inspect what SQL was executed

Example structure:
```javascript
// tests/helpers/db.mjs
let _responses = [];
let _calls = [];

export function setupMock(responses) {
  _responses = responses;
  _calls = [];
}

export function resetMock() {
  _responses = [];
  _calls = [];
}

export function getCalls() {
  return [..._calls];
}

export async function mockQuery(text, params) {
  _calls.push({ text, params });
  const match = _responses.find(r => text.includes(r.match));
  if (match) return match.result;
  return { rows: [], rowCount: 0 };
}
```

The mock will be imported by the test file and injected via module mock or direct import override. Keep it simple — the test file will use `supertest` at the HTTP level so the mock may not be needed for all tests, but it must exist per VALIDATION.md requirements.
