```javascript
const ALLOWED_PATCH_FIELDS = [
  'source', 'payment_method', 'lead_stage', 'status',
  'check_in', 'check_out', 'check_in_time', 'checkout_time', 'follow_up_at',
  'guests_count', 'total_amount', 'notes', 'cabin_id',
  'nightly_rate', 'nights',
  'cleaning_supplement', 'season_type', 'reservation_document_type'
];

const updates = [];
const params = [];
for (const field of ALLOWED_PATCH_FIELDS) {
  if (field in req.body) {
    params.push(req.body[field]);
    updates.push(`${field} = $${params.length}`);
  }
}
if (updates.length === 0) {
  return res.status(400).json({ error: "no hay campos para actualizar" });
}
params.push(id);
const result = await query(
  `UPDATE reservations SET ${updates.join(', ')} WHERE id = $${params.length} RETURNING *`,
  params
);
if (result.rowCount === 0) {
  return res.status(404).json({ error: "reserva no encontrada" });
}
res.json(result.rows[0]);
```

Wrap in try/catch and call next(err) on error (same pattern as other handlers).
